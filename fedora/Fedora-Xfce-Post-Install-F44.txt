# Fedora-44-Xfce-Post-Install-Commands by rosset


# enable sshd
sudo systemctl enable sshd --now


# install my stuff
sudo dnf install -y \
cabextract calibre curl default-editor dkms filezilla fontconfig git gitk \
hexchat htop iotop libreoffice-writer lzop nosync qbittorrent redshift \
redshift-gtk rtorrent screen sysstat tig tmux unrar unzip \
vim-default-editor vim-enhanced vlc wget wireshark wireshark-cli \
xorg-x11-font-utils zstd


# update your fresh installation
sudo dnf update -y
sudo reboot


# install dev packages & group packages
sudo dnf group install -y "c-development"
sudo dnf group install -y "cloud-infrastructure"
sudo dnf group install -y "cloud-management"
sudo dnf group install -y "container-management"
sudo dnf group install -y "development-tools"
sudo dnf group install -y "rpm-development-tools"
sudo dnf group install -y "system-tools"
sudo dnf group install "libreoffice" -y


# for postgresql buildfarm
sudo dnf builddep postgresql -y
sudo dnf install -y clang clang-tools-extra llvm-devel numactl-devel \
perl-FindBin perl-IPC-Run perl-LWP-Protocol-https perl-Test-Simple \
perl-OpCode libcurl-devel liburing-devel tcl-devel tk-devel


# fedora packaging stuff
sudo dnf install -y cmake fedora-easy-karma fedora-packager meson nosync \
python3-rpkg rpkg rpkg-common rpkg-macros valgrind


# debug tools
sudo dnf install crash kexec-tools kdump-utils -y
sudo kdumpctl reset-crashkernel
sudo grub2-mkconfig -o /boot/grub2/grub.cfg
sudo systemctl enable kdump --now
sudo kdumpctl status
